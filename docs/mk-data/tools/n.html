# ubuntu安装

  • 首先安装npm包管理工具
sudo apt install npm
  • 安装
sudo npm install -g n
  • 验证是否安装成功
n -V
  • 安装node
n lts # n 10.16.0

# 安装node版本

n stable # 安装稳定的官方版本
n lts # 安装最新的LTS官方版本
n latest # 安装最新的node版本
n <version> # 安装指定node版本

# 列出可用的版本

n ls-remote

# 列出安装的node

n ls

# node版本切换

n # 上下箭头选择，回车切换

# 卸载node

n rm <version> # n - <version> 删除缓存版本
n uninstall <version> # 删除node&npm
n prune # 删除除当前版本外所有其他版本
