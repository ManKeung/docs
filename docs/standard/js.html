# 注释规范强烈推荐

提示

好的注释使代码可读性更强。

// 单行注释
/**
 * 多行注释
 * /

正面例子

// 注释信息很多写在变量上方
let name = 'mkimq' // 注释信息少

// 函数参数超过2个或以上

/**
 * @Author: Mankeung
 * @description: 方法描述
 * @param {string} name 参数描述
 * @param {number} age 参数描述
 * @return {void} 返回
 */
function getUser(name, age) {
    ...
}

// 函数参数少于2个

// 方法描述 name [string] 参数描述
function getUser(name) {
    ...
}

// 方法描述
function  getUser() {
    ...
}
