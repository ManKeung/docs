# 10дёӘJavaScript Promiseзҡ„йқўиҜ•йўҳ

PromiseжҳҜJavaScriptејӮжӯҘзј–зЁӢзҡ„е…ій”®зү№жҖ§гҖӮдёҚз®ЎдҪ зҲұе®ғиҝҳжҳҜжҒЁе®ғпјҢдҪ йғҪеҝ…йЎ»зҗҶи§Је®ғгҖӮ

еңЁиҝҷйҮҢпјҢжҲ‘ж•ҙзҗҶдәҶдёҖдәӣе…ідәҺPromiseзҡ„йқўиҜ•жҢ‘жҲҳйўҳпјҢд»ҺеҹәзЎҖеҲ°й«ҳзә§гҖӮжңү10дёӘд»Јз ҒзүҮж®өпјҢеңЁйҳ…иҜ»жҲ‘зҡ„еҲҶжһҗд№ӢеүҚпјҢиҜ·е…ҲиҮӘе·ұе°қиҜ•дёҖдёӢгҖӮ

# еҗҢжӯҘд»Јз Ғеқ—

console.log('start')

const p1 = new Promise((resolve, reject) => {
    console.log(1)
})

console.log('end')

з¬¬дёҖдёӘй—®йўҳйқһеёёз®ҖеҚ•гҖӮ

  • жҲ‘д»¬зҹҘйҒ“пјҡ
    • еҗҢжӯҘзҡ„д»Јз Ғеқ—жҖ»жҳҜд»ҺдёҠеҲ°дёӢйЎәеәҸжү§иЎҢгҖӮ
    • еҪ“жҲ‘д»¬и°ғз”Ё new Promise(callback) ж—¶пјҢеӣһи°ғеҮҪж•°дјҡз«ӢеҚіжү§иЎҢгҖӮ

жүҖд»Ҙиҝҷж®өд»Јз ҒжҳҜдҫқж¬Ўиҫ“еҮәstartгҖҒ1гҖҒendгҖӮ

# еҮәзҺ°ејӮжӯҘд»Јз Ғзҡ„ең°ж–№

console.log('start')

const p1 = new Promise((resolve, reject) => {
    console.log(1)
    resolve(2)
})

p1.then(res => {
    console.log(res)
})

console.log('end')

еңЁиҝҷж®өд»Јз ҒзүҮж®өдёӯпјҢеҮәзҺ°дәҶдёҖж®өејӮжӯҘд»Јз ҒгҖӮд№ҹе°ұжҳҜ.then()дёӯзҡ„еӣһи°ғеҮҪж•°гҖӮ

иҜ·и®°дҪҸпјҢJavaScriptеј•ж“ҺжҖ»жҳҜе…Ҳжү§иЎҢеҗҢжӯҘд»Јз ҒпјҢз„¶еҗҺеҶҚжү§иЎҢејӮжӯҘд»Јз ҒгҖӮ

йҒҮеҲ°иҝҷдёӘй—®йўҳпјҢжҲ‘д»¬еҸӘйңҖиҰҒеҢәеҲҶеҗҢжӯҘд»Јз Ғе’ҢејӮжӯҘд»Јз ҒеҚіеҸҜгҖӮ

жүҖд»Ҙиҫ“еҮәзҡ„з»“жһңжҳҜstartгҖҒ 1гҖҒendе’Ң2гҖӮ

# жңүвҖңresolveвҖқзҡ„ең°ж–№

console.log('start')

const p1 = new Promise((resolve, reject) => {
    console.log(1)
    resolve(2)
    console.log(3)
})

p1.then(res => {
    console.log(res)
})

console.log('end')

иҝҷж®өд»Јз ҒдёҺеүҚйқўзҡ„д»Јз ҒеҮ д№ҺзӣёеҗҢпјӣе”ҜдёҖзҡ„еҢәеҲ«жҳҜеңЁresolve(2)д№ӢеҗҺжңүдёҖдёӘconsole.log(3)гҖӮ

иҜ·и®°дҪҸпјҢresolve ж–№жі•дёҚдјҡдёӯж–ӯеҮҪж•°зҡ„жү§иЎҢгҖӮе®ғиғҢеҗҺзҡ„д»Јз Ғд»Қз„¶дјҡз»§з»ӯжү§иЎҢгҖӮ

жүҖд»Ҙиҫ“еҮәз»“жһңжҳҜstart, 1, 3, endе’Ң2гҖӮ

еӣ дёәжҲ‘йҒҮеҲ°иҝҮдёҖдәӣдәәи®Өдёәresolveдјҡдёӯж–ӯеҮҪж•°зҡ„жү§иЎҢпјҢжүҖд»ҘеңЁиҝҷйҮҢејәи°ғдёҖдёӢгҖӮ

# дёҚиў«и°ғз”ЁвҖңresolveвҖқзҡ„ең°ж–№

console.log('start')

const p1 = new Promise((resolve, reject) => {
    console.log(1)
})

p1.then(res => {
    console.log(2)
})

console.log('end')

еңЁиҝҷж®өд»Јз ҒдёӯпјҢresolveж–№жі•д»ҺжңӘиў«и°ғз”ЁиҝҮпјҢеӣ жӯӨpromise1е§Ӣз»ҲеӨ„дәҺжҢӮиө·зҠ¶жҖҒгҖӮжүҖд»Ҙpromise1.then(...)д»ҺжңӘиў«жү§иЎҢиҝҮгҖӮ2дёҚдјҡеңЁжҺ§еҲ¶еҸ°дёӯжү“еҚ°еҮәжқҘгҖӮ

жүҖд»Ҙиҫ“еҮәз»“жһңжҳҜstart, 1, endгҖӮ

# и®©дҪ еӣ°жғ‘зҡ„ең°ж–№

console.log('start')

const fn = () => new Promise((resolve, reject) => {
    console.log(1)
    resolve('success')
})

console.log('middle')

fn().then(res => {
    console.log(res)
})

console.log('end')

иҝҷж®өд»Јз Ғзү№ж„ҸеўһеҠ дәҶдёҖдёӘиҝ·жғ‘жҢ‘жҲҳиҖ…зҡ„еҠҹиғҪпјҢйӮЈе°ұжҳҜfnгҖӮ

дҪҶжҳҜиҜ·и®°дҪҸпјҢж— и®әжңүеӨҡе°‘еұӮеҮҪж•°и°ғз”ЁпјҢжҲ‘д»¬зҡ„еҹәжң¬еҺҹеҲҷйғҪжҳҜдёҖж ·зҡ„пјҡ

е…Ҳжү§иЎҢеҗҢжӯҘд»Јз ҒпјҢеҶҚжү§иЎҢејӮжӯҘд»Јз Ғ

еҗҢжӯҘд»Јз ҒжҢүи°ғз”ЁйЎәеәҸжү§иЎҢ

жүҖд»Ҙиҫ“еҮәз»“жһңжҳҜstartгҖҒmiddleгҖҒ1гҖҒendе’ҢsuccessгҖӮ

еҘҪзҡ„пјҢдҪ и§үеҫ—иҝҷдәӣжҢ‘жҲҳе®№жҳ“еҗ—пјҹ

дәӢе®һдёҠпјҢиҝҷдәӣеҸӘжҳҜејҖиғғиҸңгҖӮPromiseзҡ„йҡҫзӮ№еңЁдәҺе®ғдёҺsetTimeoutдёҖиө·еҮәзҺ°гҖӮжҺҘдёӢжқҘпјҢжҲ‘д»¬еҠ еӨ§жҢ‘жҲҳзҡ„йҡҫеәҰгҖӮ

дҪ еҮҶеӨҮеҘҪдәҶеҗ—пјҹејҖеҸ‘иҖ…пјҒ

# Fulfilling Promise

console.log('start')

Promise.resolve(1).then(res => {
    console.log(res)
})

Promise.resolve(2).then(res => {
    console.log(res)
})

console.log('end')

иҝҷйҮҢPromise.resolve(1)е°Ҷиҝ”еӣһдёҖдёӘPromiseеҜ№иұЎпјҢе…¶зҠ¶жҖҒдёәе·Іе®ҢжҲҗпјҢз»“жһңдёә1пјҢе®ғжҳҜеҗҢжӯҘд»Јз ҒгҖӮ

жүҖд»Ҙиҫ“еҮәз»“жһңжҳҜstartгҖҒendгҖҒ1е’Ң2гҖӮ

# жөӢиҜ•еӣһи°ғеҹәзЎҖзҡ„ең°ж–№

console.log('start')

setTimeout(() => {
    console.log('setTimeout')
})

Promise.resolve().then(() => {
    console.log('resolve')
})

console.log('end')

жіЁж„ҸпјҒ

иҝҷжҳҜдёҖдёӘйқһеёёеӣ°йҡҫзҡ„й—®йўҳгҖӮеҰӮжһңдҪ иғҪжӯЈзЎ®еӣһзӯ”иҝҷдёӘй—®йўҳе№¶иҜҙжҳҺеҺҹеӣ пјҢйӮЈд№ҲдҪ еҜ№ JavaScript дёӯејӮжӯҘзј–зЁӢзҡ„зҗҶи§Је·Із»ҸиҫҫеҲ°дәҶдёӯзә§ж°ҙе№ігҖӮ

еңЁи§ЈйҮҠиҝҷдёӘй—®йўҳд№ӢеүҚпјҢи®©жҲ‘д»¬е…Ҳи®Ёи®әдёҖдёӢзӣёе…ізҡ„зҗҶи®әеҹәзЎҖзҹҘиҜҶгҖӮ

д№ӢеүҚжҲ‘д»¬иҜҙиҝҮеҗҢжӯҘд»Јз ҒжҳҜжҢүз…§и°ғз”ЁйЎәеәҸжү§иЎҢзҡ„пјҢйӮЈд№ҲиҝҷдәӣејӮжӯҘеӣһи°ғеҮҪж•°жҳҜжҢүз…§д»Җд№ҲйЎәеәҸжү§иЎҢзҡ„е‘ўпјҹ

жңүдәәеҸҜиғҪдјҡиҜҙпјҢи°Ғе…Ҳе®ҢжҲҗи°Ғе…Ҳжү§иЎҢгҖӮе—ҜпјҢиҝҷжҳҜзңҹзҡ„пјҢдҪҶжҳҜеҰӮжһңдёӨдёӘејӮжӯҘд»»еҠЎеҗҢж—¶е®ҢжҲҗе‘ўпјҹ

жҜ”еҰӮдёҠйқўзҡ„д»Јз ҒдёӯпјҢsetTimeoutзҡ„е®ҡж—¶еҷЁжҳҜ0з§’пјҢPromise.resolve()д№ҹдјҡеңЁжү§иЎҢеҗҺз«ӢеҚіиҝ”еӣһдёҖдёӘе·Іе®ҢжҲҗзҡ„PromiseеҜ№иұЎгҖӮ

дёӨдёӘејӮжӯҘд»»еҠЎйғҪжҳҜз«ӢеҚіе®ҢжҲҗзҡ„пјҢйӮЈд№Ҳи°Ғзҡ„еӣһи°ғеҮҪж•°дјҡе…Ҳжү§иЎҢе‘ўпјҹ

жңүзҡ„дәәеҸҜиғҪдјҡиҜҙsetTimeoutеңЁеүҚйқўпјҢжүҖд»Ҙе…Ҳжү“еҚ°setTimeoutпјҢеҶҚжү“еҚ°resolveгҖӮе®һйҷ…дёҠпјҢиҝҷз§ҚиҜҙжі•жҳҜй”ҷиҜҜзҡ„гҖӮ

жҲ‘д»¬зҹҘйҒ“еҫҲеӨҡдәӢжғ…дёҚжҳҜжҢүз…§е…Ҳиҝӣе…ҲеҮәзҡ„йЎәеәҸжү§иЎҢзҡ„пјҢжҜ”еҰӮжөҒйҮҸгҖӮ

  • еңЁJavaScript EventLoopдёӯпјҢиҝҳжңүдјҳе…Ҳзә§зҡ„жҰӮеҝөгҖӮ
    • е…·жңүиҫғй«ҳдјҳе…Ҳзә§зҡ„д»»еҠЎз§°дёәеҫ®д»»еҠЎгҖӮеҢ…жӢ¬пјҡPromiseгҖҒObjectObserverгҖҒMutationObserverгҖҒprocess.nextTickгҖҒasync/awaitгҖӮ
    • дјҳе…Ҳзә§иҫғдҪҺзҡ„д»»еҠЎз§°дёәе®Ҹд»»еҠЎгҖӮеҢ…жӢ¬пјҡsetTimeoutгҖҒsetIntervalе’ҢXHRгҖӮ

иҷҪз„¶setTimeoutе’ҢPromise.resolve()жҳҜеҗҢж—¶е®ҢжҲҗзҡ„пјҢз”ҡиҮіsetTimeoutзҡ„д»Јз ҒиҝҳжҳҜйўҶе…Ҳзҡ„пјҢдҪҶжҳҜз”ұдәҺе®ғзҡ„дјҳе…Ҳзә§дҪҺпјҢеұһдәҺе®ғзҡ„еӣһи°ғеҮҪж•°жҳҜеңЁеҗҺйқўжү§иЎҢзҡ„гҖӮ

жүҖд»Ҙиҫ“еҮәз»“жһңжҳҜstartгҖҒendгҖҒresolveе’ҢsuccessгҖӮ

# дёҖдёӘжңүеҫ®д»»еҠЎе’Ңе®Ҹд»»еҠЎд»Јз Ғзҡ„

const promise = new Promise((resolve, reject) => {
    console.log(1)

    setTimeout(() => {
        console.log('timerStart')
        resolve('success')
        console.log('timerEnd')
    })

    console.log(2)
})

promise.then(res => {
    console.log(res)
})

console.log(4)

еҰӮжһңжӮЁе·Із»ҸдәҶи§ЈдәҶеүҚйқўзҡ„д»Јз ҒзүҮж®өпјҢйӮЈд№ҲиҝҷдёӘжҢ‘жҲҳеҫҲе®№жҳ“е®ҢжҲҗгҖӮ

  • жҲ‘д»¬еҸӘйңҖиҰҒеҒҡдёүдёӘжӯҘйӘӨпјҡ

    • жүҫеҲ°еҗҢжӯҘд»Јз ҒгҖӮ
    • жүҫеҲ°еҫ®д»»еҠЎд»Јз Ғ
    • жүҫеҲ°е®Ҹд»»еҠЎд»Јз Ғ

  • йҰ–е…ҲпјҢжү§иЎҢеҗҢжӯҘд»Јз Ғпјҡ

    • иҫ“еҮә1гҖҒ2е’Ң4гҖӮ

  • з„¶еҗҺжү§иЎҢеҫ®д»»еҠЎпјҡ

дҪҶжҳҜиҝҷйҮҢжңүдёӘйҷ·йҳұпјҡз”ұдәҺеҪ“еүҚзҡ„promiseиҝҳеӨ„дәҺpendingзҠ¶жҖҒпјҢжүҖд»ҘиҝҷйҮҢзҡ„д»Јз ҒжҡӮж—¶дёҚдјҡиў«жү§иЎҢгҖӮ

  • з„¶еҗҺжү§иЎҢе®Ҹд»»еҠЎпјҡ

е№¶дё”promiseзҡ„зҠ¶жҖҒжӯЈеңЁжҲҗдёәfulfilledгҖӮ

  • з„¶еҗҺпјҢдҪҝз”ЁдәӢд»¶еҫӘзҺҜпјҢеҶҚж¬Ўжү§иЎҢеҫ®д»»еҠЎпјҡ

жүҖд»Ҙжү§иЎҢж•Ҳжһңпјҡ1гҖҒ4гҖҒtimerStartгҖҒtimerEndгҖҒsuccessгҖӮ

# е°Ҷиў«иҰҒжұӮеңЁеҫ®д»»еҠЎе’Ңе®Ҹд»»еҠЎд№Ӣй—ҙиҝӣиЎҢдјҳе…ҲжҺ’еәҸ

еңЁд»Ӣз»Қеҫ®д»»еҠЎе’Ңе®Ҹд»»еҠЎзҡ„дјҳе…Ҳзә§д№ӢеүҚпјҢиҝҷйҮҢе…ҲзңӢдёҖдёӢеҫ®д»»еҠЎе’Ңе®Ҹд»»еҠЎдәӨжӣҝжү§иЎҢзҡ„жғ…еҶөгҖӮ

const timer1 = setTimeout(() => {
    console.log('timer1')

    const p1 = Promise.resolve().then(() => {
        console.log('p1')
    })
}, 0)

const timer2 = setTimeout(() => {
    console.log('timer2')
})

иҫ“еҮәпјҡtimer1гҖҒp1гҖҒtimer2

# жңҖеҗҺдёҖж¬ЎжөӢиҜ•дҪ зҡ„PromiseеҹәзЎҖ

иҝҷжҳҜжҲ‘д»¬жңҖеҗҺдёҖдёӘжҢ‘жҲҳпјҢеҰӮжһңдҪ иғҪжӯЈзЎ®иҜҙеҮәиҝҷж®өд»Јз Ғзҡ„иҫ“еҮәз»“жһңпјҢйӮЈд№ҲдҪ еҜ№Promiseзҡ„зҗҶи§Је°ұе·Із»ҸеҫҲејәдәҶпјҢеҗҢзұ»еһӢзҡ„йқўиҜ•йўҳж №жң¬йҡҫдёҚеҖ’дҪ гҖӮ

console.log('start')

const p1 = Promise.resolve().then(() => {
    console.log('p1')

    const t2 = setTimeout(() => {
        console.log('t2')
    }, 0)
})

const t1 = setTimeout(() => {
    console.log('t1')
    const p2 = Promise.resolve().then(() => {
        console.log('p2')
    })
}, 0)
