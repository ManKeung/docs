# 断言使用

# Enums 枚举

不设置值时，值以0开始递增

enum SexType {
    BOY,
    GIRL
}

当某个字段设置数值类型的值后，后面的在它基础上递增 可以将值设置为其他类型

# as断言

function fn(arg: number) {
	return arg ? 'mkimq' : 2039
}

let f = fn(1) as string // let f: string

function fn(arg: number) {
	return arg ? 'mkimq' : 2039
}

let f = <string>fn(1) // let f: string

# const断言

# let & const

  • const 保证该字面量的严格类型
  • let 为通用类型比如字符串类型
const m = 'mkimq' // cconst m: 'mkimq'
let mk = 'mkimq' // let mk: string

# const

const断言告诉编译器为表达式推断出它能推断出的最窄或最特定的类型，而不是宽泛的类型

  • 字符串、布尔类型转换为具体值
  • 对象转换为只读属性
  • 数组转换成为只读元组
let user = {name: 'mkimq'} as const
user.name = 'mankueng' // 因为是只读属性，所以不允许设置值

# 数组赋值

变量 f 得到的类型是数组的类型 string|number，所以只要值是这两个类型都可以

let a = 'mkimq'
let b = 100

let m = [a, b] // let m: (string | number)[]
let f = m[1] // let f: string | number

// 使用 const 后会得到值的具体类型，面不是数组的类型
let m = [a, b] as const // let m: readonly [string, number]
// 上面也可以使用这种写法
let m = <const>[a, b]

let f = m[1] // let f: number
f = 'mkimq' // 报错 只能是number

# 解构

解构得到的变量类型不是具体类型

function fn() {
	let a = 'mkimq.com'
	let b = (x: number, y: number): number => x + y

	return [a, b]
}

let [n, m] = fn() // 变量 m 的类型为 string | (() => void)
m(1, 2) // 报错：因为类型可能是字符串，所以不允许调用

// ok
console.log((m as Function)(1, 2))
console.log((m as (x: number, y: number) => number)(1, 5))

// 调用时对返回值断言类型
let [n, m] = fn() as [string, (x: number, y: number) => number]

可以在函数体内声明返回类型

function fn() {
	let a = 'mkimq.com'
	let b = (x: number, y: number): number => x + y

	return [a, b] as [typeof a, typeof b]
}

使用 as const 就可以很高效的解决上面的问题，可以得到具体的类型，来得到更安全的代码，同时会有更好的代码提示

function fn() {
	let a = 'mkimq.com'
	let b = (x: number, y: number): number => x + y

	return [a, b] as const
}

因为 const 是取值的类型，下面代码虽然不报错，但此时 b 的类型已经是 字符串或函数，所以像下面一样在函数调用时 as const 没有意义

function fn() {
	let a = 'mkimq.com'
	let b = (x: number, y: number): number => x + y

	return [a, b]
}

const [n, m] = [...fn()] as const

# null/undefined

默认情况下 null 与undefined 可以赋值给其他类型

let m: string = 'mkimq'
m = null
m = undefined

当我们修改 tsconfig.json 配置文件的strictNullChecks 字段为 true（默认即为 true） 时，则不能将 null、undefined 赋值给其他类型

"strictNullChecks": true

除非向下面一样明确指定类型

let m: string | undefiend | null = 'mkimq'

# 非空断言

下面的示例获取的值可能为HTMLDivElement或null，所以直接分配类型“HTMLDivElement”将报错误

下例操作需要开启 tsconfig.json 的配置项strictNullChecks 字段为 true

const el: HTMLDivElement = document.querySelector('.mk')
console.log(el.id)

可以使用 as 断言类型

const el: HTMLDivElement = document.querySelector('.mk') as HTMLDivElement
console.log(el.id)

在值后面使用 ! 来声明值非null

const el: HTMLDivElement = document.querySelector('.mk')!
console.log(el.id)

# 类型推断

对于获取的标签对象可能是为 null 也可能是该标签类型

  • body等具体标签可以准确标签类型或 null
  • 根据class 等获取不能准确获取标签类型，推断的类型为 Element|null
const body = document.querySelector('body') // const body: HTMLBodyElement|null

根据 class 获取标签结果是 ELement 并不是具体的标签，因为根据 class无法确定标签类型

const el = document.querySelector('.mk') // const el: Element | null

# null处理

针对于其他标签元素，返回值可能为null，所以使用 as 断言或！处理

let div = document.querySelector('div') as HTMLDivElement // const div: HTMLDivElement
// 或使用
div = document.querySelector('div')! // 非空断言
console.log(div.id)

# 断言处理

使用as 将类型声明为 HTMLAnchorElement 则 TS 会将其按 a 链接类型处理

现在所有的提示将是 a 链接属性或方法

const el = document.querySelector('.mk') as HTMLAnchorElement // const el: HTMLAnchorElement
console.log(el.href)

# 事件处理

下面提取按钮元素并添加事件，实现插入元素的功能

const body = document.querySelector('body')
const bt = document.querySelector('#bt') as HTMLButtonElement

bt.addEventListener('click', (e: Event) => {
	e.preventDefault() // 因为设置了 e 的类型，所以会有完善的提示
	body.insertAdjacentHTML('beforeend', '<div>mkimq</div>')
})
